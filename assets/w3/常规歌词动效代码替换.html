      if (savedSettings.fontVersion !== 2) {
        savedSettings.fontChoice = "sourceSerif4";
        savedSettings.fontVersion = 2;
      }
      if (savedSettings.floatVersion !== 3) {
        savedSettings.floatDistance = 6;
        savedSettings.floatDuration = 2600;
        if (!savedSettings.effect || savedSettings.effect === "shine") {
          savedSettings.effect = "shine-float";
        }
        savedSettings.floatVersion = 3;
      }
      let settings = { ...defaultSettings, ...savedSettings };
      settings.fontChoice = fontStacks[settings.fontChoice] ? settings.fontChoice : defaultSettings.fontChoice;
      settings.fontColor = normalizeHexColor(settings.fontColor, defaultSettings.fontColor);
      settings.effect = ["shine", "wave", "shine-wave", "float", "shine-float", "wave-float", "shine-wave-float", "none"].includes(settings.effect)
        ? settings.effect
        : defaultSettings.effect;
      settings.fallStyle = ["quick", "soft"].includes(settings.fallStyle) ? settings.fallStyle : defaultSettings.fallStyle;
      settings.dropDuration = Math.max(300, Math.min(12000, settings.dropDuration));
      settings.fallDuration = Math.max(200, Math.min(6000, settings.fallDuration));
      settings.fallFadeDuration = Math.max(100, Math.min(3000, settings.fallFadeDuration ?? defaultSettings.fallFadeDuration));
      settings.groupFallGap = Math.max(0, Math.min(1500, settings.groupFallGap ?? defaultSettings.groupFallGap));
      settings.lineFallGap = Math.max(0, Math.min(2000, settings.lineFallGap ?? defaultSettings.lineFallGap));
      settings.groupFallSpread = Math.max(0, Math.min(320, settings.groupFallSpread ?? defaultSettings.groupFallSpread));
      let index = 0;
      let timerIds = [];
      let restartRequest = 0;

      function loadSettings() {
        try {
          return JSON.parse(window.localStorage.getItem("lyricControls") || "{}");
        } catch {
          return {};
        }
      }

      function saveSettings() {
        window.localStorage.setItem("lyricControls", JSON.stringify(settings));
      }

      function resizeStage() {
        const width = wrap.getBoundingClientRect().width;
        stage.style.transform = `scale(${width / 720})`;
      }

      function formatValue(key, value) {
        if (key === "effect" || key === "fontChoice" || key === "fallStyle") {
          return "";
        }
        if (key === "fontColor") {
          return normalizeHexColor(value, defaultSettings.fontColor).toUpperCase();
        }
        if (key.includes("Duration")) {
          return `${(value / 1000).toFixed(1)}s`;
        }
        if (key === "groupFallGap") {
          return `${value}ms`;
        }
        if (key === "lineFallGap") {
          return `${value}ms`;
        }
        if (key === "groupFallSpread") {
          return `${value}px`;
        }
        if (key === "floatDistance") {
          return `${value}px`;
        }
        if (key === "trailLength") {
          return `${Number(value).toFixed(1)}x`;
        }
        if (key === "trailOpacity") {
          return `${Math.round(Number(value) * 100)}%`;
        }
        if (key === "fontOpacity") {
          return `${Math.round(Number(value) * 100)}%`;
        }
        return `${value}px`;
      }

      function clampAlpha(value) {
        return Math.max(0, Math.min(1, value)).toFixed(3);
      }

      function normalizeHexColor(value, fallback) {
        return /^#[0-9a-f]{6}$/i.test(value || "") ? value.toLowerCase() : fallback;
      }

      function hexToRgb(hex) {
        const normalized = normalizeHexColor(hex, defaultSettings.fontColor).slice(1);
        return {
          r: parseInt(normalized.slice(0, 2), 16),
          g: parseInt(normalized.slice(2, 4), 16),
          b: parseInt(normalized.slice(4, 6), 16),
        };
      }

      function rgbToHex({ r, g, b }) {
        return `#${[r, g, b].map((channel) => channel.toString(16).padStart(2, "0")).join("")}`;
      }

      function mixWithWhite(hex, amount) {
        const rgb = hexToRgb(hex);
        return rgbToHex({
          r: Math.round(rgb.r + (255 - rgb.r) * amount),
          g: Math.round(rgb.g + (255 - rgb.g) * amount),
          b: Math.round(rgb.b + (255 - rgb.b) * amount),
        });
      }

      function applySettings() {
        const root = document.documentElement.style;
        const fontRgb = hexToRgb(settings.fontColor);
        root.setProperty("--lyric-left", `${settings.left}px`);
        root.setProperty("--lyric-top", `${settings.top}px`);
        root.setProperty("--lyric-width", `${settings.width}px`);
        root.setProperty("--lyric-font-size", `${settings.fontSize}px`);
        root.setProperty("--lyric-font-family", fontStacks[settings.fontChoice]);
        root.setProperty("--lyric-font-weight", fontWeights[settings.fontChoice]);
        root.setProperty("--lyric-color", settings.fontColor);
        root.setProperty("--lyric-color-rgb", `${fontRgb.r} ${fontRgb.g} ${fontRgb.b}`);
        root.setProperty("--lyric-shine-edge", `rgb(${fontRgb.r} ${fontRgb.g} ${fontRgb.b} / 0.82)`);
        root.setProperty("--lyric-shine-mid", mixWithWhite(settings.fontColor, 0.68));
        root.setProperty("--lyric-shine-soft", mixWithWhite(settings.fontColor, 0.58));
        root.setProperty("--lyric-shine-edge-strong", `rgb(${fontRgb.r} ${fontRgb.g} ${fontRgb.b} / 0.86)`);
        root.setProperty("--lyric-shine-glow", `rgb(${fontRgb.r} ${fontRgb.g} ${fontRgb.b} / 0.24)`);
        root.setProperty("--lyric-opacity", settings.fontOpacity);
        root.setProperty("--lyric-line-height", `${settings.lineHeight}px`);
        root.setProperty("--drop-duration", `${settings.dropDuration}ms`);
        root.setProperty("--fall-duration", `${settings.fallDuration}ms`);
        root.setProperty("--fall-fade-duration", `${settings.fallFadeDuration}ms`);
        root.setProperty("--fall-fade-start", `${Math.max(0, settings.fallDuration - settings.fallFadeDuration)}ms`);
        root.setProperty("--effect-duration", `${settings.effectDuration}ms`);
        root.setProperty("--float-distance", `${settings.floatDistance}px`);
        root.setProperty("--float-up-distance", `${-settings.floatDistance}px`);
        root.setProperty("--float-down-distance", `${Math.round(settings.floatDistance * 0.42)}px`);
        root.setProperty("--float-duration", `${settings.floatDuration}ms`);
        root.setProperty("--trail-entry-long", `${-32 * settings.trailLength}px`);
        root.setProperty("--trail-entry-short", `${-16 * settings.trailLength}px`);
        root.setProperty("--trail-fall-long", `${-48 * settings.trailLength}px`);
        root.setProperty("--trail-fall-mid", `${-28 * settings.trailLength}px`);
        root.setProperty("--trail-fall-short", `${-12 * settings.trailLength}px`);
        root.setProperty("--trail-end-long", `${-52 * settings.trailLength}px`);
        root.setProperty("--trail-end-mid", `${-30 * settings.trailLength}px`);
        root.setProperty("--trail-end-short", `${-14 * settings.trailLength}px`);
        root.setProperty("--trail-entry-soft-alpha", clampAlpha(0.12 * settings.trailOpacity));
        root.setProperty("--trail-entry-strong-alpha", clampAlpha(0.2 * settings.trailOpacity));
        root.setProperty("--trail-fall-soft-alpha", clampAlpha(0.1 * settings.trailOpacity));
        root.setProperty("--trail-fall-mid-alpha", clampAlpha(0.18 * settings.trailOpacity));
        root.setProperty("--trail-fall-strong-alpha", clampAlpha(0.2 * settings.trailOpacity));
        root.setProperty("--trail-end-soft-alpha", clampAlpha(0.08 * settings.trailOpacity));
        root.setProperty("--trail-end-mid-alpha", clampAlpha(0.14 * settings.trailOpacity));
        root.setProperty("--trail-end-strong-alpha", clampAlpha(0.17 * settings.trailOpacity));

        controls.forEach((control) => {
          const key = control.dataset.control;
          control.value = settings[key];
        });
        valueLabels.forEach((label) => {
          const key = label.dataset.valueFor;
          label.textContent = formatValue(key, settings[key]);
        });
        colorPreview.textContent = settings.fontColor.toUpperCase();
        colorPreview.style.background = `linear-gradient(90deg, ${settings.fontColor}, ${mixWithWhite(settings.fontColor, 0.68)})`;
      }

      function clearTimers() {
        timerIds.forEach((id) => window.clearTimeout(id));
        timerIds = [];
      }

      function restartAnimation() {
        window.clearTimeout(restartRequest);
        clearTimers();
        lyricRoot.innerHTML = "";
        index = 0;
        playCycle();
      }

      function queueRestart() {
        window.clearTimeout(restartRequest);
        restartRequest = window.setTimeout(restartAnimation, 220);
      }

      function renderLyric(group, lines) {
        let wordIndex = 0;
        let charIndex = 0;
        const entryScale = settings.dropDuration / defaultEntryTimeline.total;
        const secondLineDelay = defaultEntryTimeline.secondLineDelay * entryScale;
        const wordStagger = defaultEntryTimeline.wordStagger * entryScale;
        const wordDropDuration = Math.max(120, defaultEntryTimeline.wordDuration * entryScale);
        group.innerHTML = lines
          .map((line, lineIndex) => {
            const words = line
              .map((word, wordInLine) => {
                const lineEntryDelay = lineIndex === 0 ? 0 : secondLineDelay;
                const delay = lineEntryDelay + wordInLine * wordStagger;
                const lineExitDelay = lineIndex === 1 ? 0 : settings.lineFallGap;
                const groupBreak = line.length <= 3 ? line.length : 2;
                const groupIndex = wordInLine < groupBreak ? 0 : 1;
                const groupStart = groupIndex === 0 ? 0 : groupBreak;
                const outDelay = lineExitDelay + groupIndex * settings.groupFallGap + (wordInLine - groupStart) * 36;
                const lineBaseY = settings.top + lineIndex * settings.lineHeight;
                const groupDistanceMap = lineIndex === 0 ? [0.2, 0.85] : [1.1, 0.55];
                const distanceOffset = groupDistanceMap[groupIndex] ?? groupDistanceMap[0];
                const redEntryY = settings.fallTargetY + distanceOffset * settings.groupFallSpread;
                const petalY = redEntryY - lineBaseY;
                const petalEndY = petalY + 46 + (wordInLine % 2) * 18;
                const letters = [...word]
                  .map((char) => {
                    const effectDelay = charIndex * 20 + lineIndex * 110;
                    charIndex += 1;
                    return `<span class="char" style="--effect-delay:${effectDelay}ms">${char}</span>`;
                  })
                  .join("");
                wordIndex += 1;
                return `<span class="word" style="--drop-delay:${delay}ms;--word-drop-duration:${wordDropDuration}ms;--out-delay:${outDelay}ms;--petal-y:${petalY}px;--petal-end-y:${petalEndY}px">${letters}</span>`;
              })
              .join("");
            const floatDelay = lineIndex === 0 ? "0ms" : "360ms";
            return `<div class="line" style="--float-delay:${floatDelay}">${words}</div>`;
          })
          .join("");
      }

      function playCycle() {
        const current = lyrics[index];
        const group = document.createElement("div");
        group.className = "lyric-group";
        lyricRoot.appendChild(group);
        renderLyric(group, current);

        const longestLine = Math.max(...current.map((line) => line.length));
        const dropDone = settings.dropDuration + 100;
        const falloutStart = dropDone + settings.holdDuration;
        const nextStart = falloutStart + 520;
        const falloutDuration = Math.max(settings.fallDuration, settings.fallFadeDuration);
        const cleanup = falloutStart + falloutDuration + settings.lineFallGap + settings.groupFallGap + longestLine * 135 + 1400;

        timerIds.push(window.setTimeout(() => {
          if (settings.effect !== "none") {
            group.classList.add(`effect-${settings.effect}`);
          }
        }, dropDone));

        timerIds.push(window.setTimeout(() => {
          group.classList.remove("effect-shine", "effect-wave", "effect-shine-wave", "effect-float", "effect-shine-float", "effect-wave-float", "effect-shine-wave-float");
          group.classList.add("fallout", `fall-style-${settings.fallStyle}`);
        }, falloutStart));

        index = (index + 1) % lyrics.length;
        timerIds.push(window.setTimeout(playCycle, nextStart));
        const cleanupId = window.setTimeout(() => {
          group.remove();
          timerIds = timerIds.filter((id) => id !== cleanupId);
        }, cleanup);
        timerIds.push(cleanupId);
      }

      controls.forEach((control) => {
        control.addEventListener("input", () => {
          const key = control.dataset.control;
          settings[key] = control.type === "range" ? Number(control.value) : control.value;
          applySettings();
          saveSettings();
          if (["top", "lineHeight", "dropDuration", "holdDuration", "fallDuration", "fallFadeDuration", "fallStyle", "groupFallGap", "lineFallGap", "groupFallSpread", "fallTargetY", "effectDuration", "effect"].includes(key)) {
            queueRestart();
          }
        });
      });

      document.querySelectorAll("[data-fall-shift]").forEach((button) => {
        button.addEventListener("click", () => {
          const control = document.querySelector('[data-control="fallTargetY"]');
          const nextValue = settings.fallTargetY + Number(button.dataset.fallShift);
          settings.fallTargetY = Math.max(Number(control.min), Math.min(Number(control.max), nextValue));
          applySettings();
          saveSettings();
          restartAnimation();
        });
      });

      document.querySelector('[data-action="restart"]').addEventListener("click", restartAnimation);
      document.querySelector('[data-action="reset"]').addEventListener("click", () => {
        settings = { ...defaultSettings };
        applySettings();
        saveSettings();
        restartAnimation();
      });

      window.addEventListener("resize", resizeStage);
      applySettings();
      resizeStage();
      playCycle();
